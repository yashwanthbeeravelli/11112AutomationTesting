from selenium import webdriver
from selenium.webdriver.chrome.service import Service
from webdriver_manager.chrome import ChromeDriverManager
from selenium.webdriver.common.by import By
import time

driver = webdriver.Chrome(service=Service(ChromeDriverManager().install()))

driver.get("http://mail.demo.studentnet.edu.au/")

usernameTxtBox = driver.find_element(By.XPATH, "//*[@id='username']")
passwordTxtBox = driver.find_element(By.XPATH, "//*[@id='password']")
usernameTxtBox.send_keys("studentexample")
passwordTxtBox.send_keys("demo1234")

time .sleep(5)

logInBtn = driver.find_element(By.XPATH, "/html/body/div[2]/div/div[2]/form/div[4]/input")
logInBtn.click()

time .sleep(10)
